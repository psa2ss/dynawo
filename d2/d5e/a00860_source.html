427 inline void create_index_spec(index_spec_t* dest, int nridx, ...) {
428  int i;
429  va_list ap;
430  va_start(ap, nridx);
431 
432  dest->ndims = nridx;
433  dest->dim_size = size_alloc(nridx);
434  dest->index = index_alloc(nridx);
435  dest->index_type = reinterpret_cast<char*>(generic_alloc(nridx+1, sizeof(char)));
436  for (i = 0; i < nridx; ++i) {
437  dest->dim_size[i] = va_arg(ap, _index_t);
438  dest->index[i] = va_arg(ap, _index_t*);
439  dest->index_type[i] = static_cast<char>(va_arg(ap, _index_t)); /* char is cast to int by va_arg.*/
440  }
441  va_end(ap);
442 }
